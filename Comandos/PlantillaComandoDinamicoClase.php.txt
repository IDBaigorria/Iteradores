use Iteradores\Comandos\Comando;

$comando = new class implements Comando {

    {

    }

    public static function solo_desarrollo(): bool
    {
        return false; // true si solo debe existir en desarrollo
    }

    public static function descripcion(): string
    {
        return 'Descripción breve del comando dinámico.';
    }

    public static function parametros(): array
    {
        return [
            // Definir aquí los parámetros igual que en una clase estática
        ];
    }

    public static function ejemplos(): array
    {
        return [
            'dinamico:accion --bandera1',
        ];
    }

    {

        return true;
    }

    {
        return null; // o devolver un callable
    }
};

Controlador::registrar_comando_desde_instancia($comando);
